void func sort(int l[2])
var int aux[2], primero, segundo, tercero;
{
    primero = l[0];
    segundo = l[1];
    tercero = l[2];
    
    if(primero<segundo){
        if(segundo<tercero){
            aux[0] = primero;
            aux[1] = segundo;
            aux[2] = tercero;
        }
        else{
            aux[2] = segundo;
            if(primero<tercero){
                aux[0] = primero;
                aux[1] = tercero;
            }
            else{
                aux[0] = tercero;
                aux[1] = primero;
            }
        }
    }
    else{
        if(primero>tercero){
            aux[2] = primero;
            if(segundo<tercero){
                aux[0] = segundo;
                aux[1] = tercero;
            }
            else{
                aux[0] = tercero;
                aux[1] = segundo;
                }
        }
        else{
            aux[0] = segundo;
            aux[1] = primero;
            aux[2] = tercero;
        }
    }
    
    write("Lista en orden");
    aux.printArray();
}

int func find(int l[2], int n)
var int cont;
{
    cont = 0;
    while(cont<3){
        if(l[cont]==n){
            return(cont);
        }
        cont = cont + 1;
    }
    return(-1);
}
